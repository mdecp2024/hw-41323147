from browser import html
from browser import document as doc
import random
import math

# 初始化畫布
def initialize_canvas(canvas_width=400, canvas_height=400, id="brython_div1"):
    canvas = html.CANVAS(width=canvas_width, height=canvas_height)
    doc[id] <= canvas
    ctx = canvas.getContext("2d")
    return canvas, ctx

# 隨機顏色生成
def random_color_generator():
    r = random.randint(0, 255)
    g = random.randint(0, 255)
    b = random.randint(0, 255)
    return f"rgb({r}, {g}, {b})"

# 判斷點是否在圓內
def is_point_in_circle(px, py, cx, cy, r):
    return (px - cx) ** 2 + (py - cy) ** 2 <= r ** 2

# 繪製三個圓形的區域和背景
def draw_three_circles(x1, y1, r1, x2, y2, r2, x3, y3, r3):
    canvas, ctx = initialize_canvas(400, 400)

    # 顏色分配
    color1 = random_color_generator()  # 圓1的獨立區域顏色
    color2 = random_color_generator()  # 圓2的獨立區域顏色
    color3 = random_color_generator()  # 圓3的獨立區域顏色
    color_12 = random_color_generator()  # 圓1和圓2交集區域顏色
    color_13 = random_color_generator()  # 圓1和圓3交集區域顏色
    color_23 = random_color_generator()  # 圓2和圓3交集區域顏色
    color_123 = random_color_generator()  # 三個圓交集區域顏色
    color_background = "white"  # 背景顏色

    # 清空畫布
    ctx.clearRect(0, 0, canvas.width, canvas.height)

    # 掃描畫布上的每個像素點
    for py in range(0, canvas.height):
        for px in range(0, canvas.width):
            # 判斷像素是否屬於某個圓或交集區域
            in_circle1 = is_point_in_circle(px, py, x1, y1, r1)
            in_circle2 = is_point_in_circle(px, py, x2, y2, r2)
            in_circle3 = is_point_in_circle(px, py, x3, y3, r3)

            if in_circle1 and in_circle2 and in_circle3:
                # 同時在三個圓內
                ctx.fillStyle = color_123
            elif in_circle1 and in_circle2:
                # 僅在圓1和圓2內
                ctx.fillStyle = color_12
            elif in_circle1 and in_circle3:
                # 僅在圓1和圓3內
                ctx.fillStyle = color_13
            elif in_circle2 and in_circle3:
                # 僅在圓2和圓3內
                ctx.fillStyle = color_23
            elif in_circle1:
                # 僅在圓1內
                ctx.fillStyle = color1
            elif in_circle2:
                # 僅在圓2內
                ctx.fillStyle = color2
            elif in_circle3:
                # 僅在圓3內
                ctx.fillStyle = color3
            else:
                # 不屬於任何圓，屬於背景
                ctx.fillStyle = color_background

            # 填充當前像素
            ctx.fillRect(px, py, 1, 1)

# 設定三個圓的圓心和半徑
x1, y1, r1 = 150, 300, 80  # 圓1：圓心 (150, 300)，半徑 80
x2, y2, r2 = 250, 300, 80  # 圓2：圓心 (250, 300)，半徑 80
x3, y3, r3 = 200, 200, 80  # 圓3：圓心 (200, 200)，半徑 80（正上方）

# 繪製三個圓形和不同區域
draw_three_circles(x1, y1, r1, x2, y2, r2, x3, y3, r3)
